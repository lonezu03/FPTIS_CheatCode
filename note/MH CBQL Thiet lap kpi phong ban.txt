DECLARE @ThietLapMucTieuCoCauID_PhongBan nvarchar(12)
BEGIN

IF(ISNULL(@ApprovedStatusID, '') = '')
BEGIN
	SET @ApprovedStatusID = '72@73@74@76@77@78@79@80@'
END

SELECT	@EmpID = EmpID
FROM	UMS_tblUserAccount (NOLOCK)
WHERE	UserGroupID = @UserGroupID

IF @Activity = 'GetYearComBo'
BEGIN
	SELECT [Year] AS ID, [Year] AS Name, [Year]
	FROM
	(
		SELECT	PM_GoalPeriodYear AS [Year]
		FROM	dbo.PM_tblGoalPeriod (NOLOCK)
	) A
	GROUP BY A.Year
	ORDER BY A.Year DESC
END
ELSE IF @Activity = 'GetYear'
BEGIN
	SELECT YEAR(@dGetDateTime) as [Year]
END
ELSE IF @Activity = 'GetDataComboByYear'
BEGIN
	SELECT	A.ID, A.Name
	FROM	(
			SELECT	A.PM_GoalPeriodID as ID, 
					case when @LanguageID = 'VN' then A.PM_GoalPeriodVNName else A.PM_GoalPeriodName end as Name, 
					A.PM_GoalPeriodYear as Nam,
					CASE WHEN Convert(date,@dGetDateTime, 103) BETWEEN A.StartDate_PhongBan AND A.EndDate_PhongBan THEN 1 ELSE 0 END as isHieuLuc,
					EndDate_PhongBan
			FROM	dbo.PM_tblGoalPeriod (Nolock) A
			WHERE	A.PM_GoalPeriodYear = @Year
					and (A.GoalPeriodStatus is null or A.GoalPeriodStatus = 0)
			--UNION
			--SELECT '' as ID, '' Name, @Year Nam,1 isHieuLuc, GETDATE() as EndDate_CongTy
			) as A
	ORDER BY A.isHieuLuc desc, EndDate_PhongBan desc
END
ELSE IF @Activity = 'LoadComboApprovedStatus'
BEGIN
	select	A.PM_ApprovedStatusID as ID, 
			case when @LanguageID = 'VN' then A.VNName else A.Name end as Name
	from	PM_tblApprovedStatus (Nolock) A
	where A.Type = 9 AND Used = 1
			AND A.PM_ApprovedStatusID IN (72, 73, 74, 76, 77, 78, 79, 80)
	order by [Rank]
END
ELSE IF @Activity = 'Search'
BEGIN
	DECLARE @IsAdjustment BIT,
			@IsHieuLuc	BIT

	SELECT	@IsHieuLuc = case when (A.GoalPeriodStatus is null or A.GoalPeriodStatus = 0) 
			AND Convert(date,@dGetDateTime, 103) between A.StartDate_PhongBan and A.EndDate_PhongBan 
			then 1 else 0 end 
	FROM	PM_tblGoalPeriod (NOLOCK) A
	WHERE	A.PM_GoalPeriodID = @GoalPeriodID

	IF(@IsHieuLuc = 1)
	BEGIN
		if object_id('tempdb..#tmpTaoPhongBan') is not null
		drop table #tmpTaoPhongBan
	
		if object_id('tempdb..#tmpThietLapMucTieuPhongBan') is not null
			drop table #tmpThietLapMucTieuPhongBan

		SELECT	A.ID
				,A.LSCompanyID
				,A.LSLevel1ID
				,A.LSLevel2ID
				,A.LSLevel3ID
				,A.LSLevel4ID
				,A.LSLevel5ID
				,A.LSLevel6ID
				,A.LSLevel7ID
				,A.LSLevel8ID
				,A.LSLevel9ID
				,A.LSTrackingMethodID
		INTO	#tmpThietLapMucTieuPhongBan
		FROM	PM_tblSetupManagerAndCEO (NOLOCK) A
				INNER JOIN 
				(
					SELECT	LSCompanyID, LSLevel1ID, LSLevel2ID, LSLevel3ID, LSLevel4ID, 
							LSLevel5ID, LSLevel6ID, LSLevel7ID, LSLevel8ID, LSLevel9ID, MAX(EffectiveDate) AS MaxDate
					FROM	PM_tblSetupManagerAndCEO (NOLOCK)
					WHERE	Type = 2
							AND EffectiveDate <= GetDate()
					GROUP BY LSCompanyID, LSLevel1ID, LSLevel2ID, LSLevel3ID, LSLevel4ID,
								LSLevel5ID, LSLevel6ID, LSLevel7ID, LSLevel8ID, LSLevel9ID
				) AS B ON B.LSCompanyID = A.LSCompanyID
					AND	CASE WHEN B.LSLevel1ID IS NULL THEN -1 ELSE B.LSLevel1ID END = CASE WHEN A.LSLevel1ID IS NULL THEN -1 ELSE A.LSLevel1ID END
					AND	CASE WHEN B.LSLevel2ID IS NULL THEN -1 ELSE B.LSLevel2ID END = CASE WHEN A.LSLevel2ID IS NULL THEN -1 ELSE A.LSLevel2ID END
					AND	CASE WHEN B.LSLevel3ID IS NULL THEN -1 ELSE B.LSLevel3ID END = CASE WHEN A.LSLevel3ID IS NULL THEN -1 ELSE A.LSLevel3ID END
					AND	CASE WHEN B.LSLevel4ID IS NULL THEN -1 ELSE B.LSLevel4ID END = CASE WHEN A.LSLevel4ID IS NULL THEN -1 ELSE A.LSLevel4ID END
					AND	CASE WHEN B.LSLevel5ID IS NULL THEN -1 ELSE B.LSLevel5ID END = CASE WHEN A.LSLevel5ID IS NULL THEN -1 ELSE A.LSLevel5ID END
					AND	CASE WHEN B.LSLevel6ID IS NULL THEN -1 ELSE B.LSLevel6ID END = CASE WHEN A.LSLevel6ID IS NULL THEN -1 ELSE A.LSLevel6ID END
					AND	CASE WHEN B.LSLevel7ID IS NULL THEN -1 ELSE B.LSLevel7ID END = CASE WHEN A.LSLevel7ID IS NULL THEN -1 ELSE A.LSLevel7ID END
					AND	CASE WHEN B.LSLevel8ID IS NULL THEN -1 ELSE B.LSLevel8ID END = CASE WHEN A.LSLevel8ID IS NULL THEN -1 ELSE A.LSLevel8ID END
					AND	CASE WHEN B.LSLevel9ID IS NULL THEN -1 ELSE B.LSLevel9ID END = CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END
					AND B.MaxDate = A.EffectiveDate
				LEFT JOIN PM_tblThietLapMucTieuCoCau (NOLOCK) D ON D.GoalPeriodID = @GoalPeriodID 
					AND D.LSCompanyID = A.LSCompanyID 
					AND D.LSLevel1ID = A.LSLevel1ID
					AND D.LSLevel2ID = A.LSLevel2ID
					AND D.LSLevel3ID = A.LSLevel3ID
					AND	CASE WHEN D.LSLevel4ID IS NULL THEN -1 ELSE D.LSLevel4ID END = CASE WHEN A.LSLevel4ID IS NULL THEN -1 ELSE A.LSLevel4ID END
					AND	CASE WHEN D.LSLevel5ID IS NULL THEN -1 ELSE D.LSLevel5ID END = CASE WHEN A.LSLevel5ID IS NULL THEN -1 ELSE A.LSLevel5ID END
					AND	CASE WHEN D.LSLevel6ID IS NULL THEN -1 ELSE D.LSLevel6ID END = CASE WHEN A.LSLevel6ID IS NULL THEN -1 ELSE A.LSLevel6ID END
					AND	CASE WHEN D.LSLevel7ID IS NULL THEN -1 ELSE D.LSLevel7ID END = CASE WHEN A.LSLevel7ID IS NULL THEN -1 ELSE A.LSLevel7ID END
					AND	CASE WHEN D.LSLevel8ID IS NULL THEN -1 ELSE D.LSLevel8ID END = CASE WHEN A.LSLevel8ID IS NULL THEN -1 ELSE A.LSLevel8ID END
					AND	CASE WHEN D.LSLevel9ID IS NULL THEN -1 ELSE D.LSLevel9ID END = CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END
					AND D.Type = 2
		WHERE	(@LSCompanyID IS NULL OR A.LSCompanyID = @LSCompanyID)
				AND (@LSLevel1ID IS NULL OR A.LSLevel1ID = @LSLevel1ID)
				AND (@LSLevel2ID IS NULL OR A.LSLevel2ID = @LSLevel2ID)
				AND (@LSLevel3ID IS NULL OR A.LSLevel3ID = @LSLevel3ID)
				AND (@LSLevel4ID IS NULL OR A.LSLevel4ID = @LSLevel4ID)
				AND (@LSLevel5ID IS NULL OR A.LSLevel5ID = @LSLevel5ID)
				AND (@LSLevel6ID IS NULL OR A.LSLevel6ID = @LSLevel6ID)
				AND (@LSLevel7ID IS NULL OR A.LSLevel7ID = @LSLevel7ID)
				AND (@LSLevel8ID IS NULL OR A.LSLevel8ID = @LSLevel8ID)
				AND (@LSLevel9ID IS NULL OR A.LSLevel9ID = @LSLevel9ID)
				AND (A.EmpID = @EmpID OR A.NguoiNhapHo_EmpID = @EmpID)	
				AND D.ThietLapMucTieuCoCauID IS NULL
		
		--SELECT  A.ID, A.LSCompanyID, A.LSLevel1ID, A.LSLevel2ID, A.LSLevel3ID
		--		, A.LSLevel4ID, A.LSLevel5ID, A.LSLevel6ID, A.LSLevel7ID, A.LSLevel8ID, A.LSLevel9ID, A.LSTrackingMethodID
		--		,dbo.PM_fnGetIDAdjustmentKPI(1, @Year, @GoalPeriodID, A.LSCompanyID, A.LSLevel1ID, A.LSLevel2ID, A.LSLevel3ID, A.LSLevel4ID, A.LSLevel5ID, A.LSLevel6ID, A.LSLevel7ID, A.LSLevel8ID, A.LSLevel9ID, A.LSTrackingMethodID, @LanguageID, '2') as ThietLapMucTieuCoCauID
		--		,@Year as [Year], @GoalPeriodID as GoalPeriodID
		--FROM	#tmpThietLapMucTieuPhongBan (NOLOCK) A

		SELECT	A.ThietLapMucTieuCoCauID, A.[Year], A.GoalPeriodID
				, A.ID, A.LSCompanyID, A.LSLevel1ID, A.LSLevel2ID, A.LSLevel3ID
				, A.LSLevel4ID, A.LSLevel5ID, A.LSLevel6ID, A.LSLevel7ID, A.LSLevel8ID, A.LSLevel9ID, A.LSTrackingMethodID
		INTO	#tmpTaoPhongBan
		FROM	(
				SELECT  A.ID, A.LSCompanyID, A.LSLevel1ID, A.LSLevel2ID, A.LSLevel3ID
						, A.LSLevel4ID, A.LSLevel5ID, A.LSLevel6ID, A.LSLevel7ID, A.LSLevel8ID, A.LSLevel9ID, A.LSTrackingMethodID
						,dbo.PM_fnGetIDAdjustmentKPI(1, @Year, @GoalPeriodID, A.LSCompanyID, A.LSLevel1ID, A.LSLevel2ID, A.LSLevel3ID, A.LSLevel4ID, A.LSLevel5ID, A.LSLevel6ID, A.LSLevel7ID, A.LSLevel8ID, A.LSLevel9ID, A.LSTrackingMethodID, @LanguageID, '2') as ThietLapMucTieuCoCauID
						,@Year as [Year], @GoalPeriodID as GoalPeriodID
				FROM	#tmpThietLapMucTieuPhongBan (NOLOCK) A
				) as A
		WHERE	ThietLapMucTieuCoCauID <> ''

		IF EXISTS (SELECT TOP 1 1 FROM #tmpTaoPhongBan (NOLOCK))
		BEGIN
			INSERT INTO PM_tblThietLapMucTieuCoCau([Year], GoalPeriodID, ApprovedStatusID,
						LSCompanyID, LSLevel1ID, LSLevel2ID, LSLevel3ID, LSLevel4ID,
						LSLevel5ID, LSLevel6ID, LSLevel7ID, LSLevel8ID, LSLevel9ID,
						LSTrackingMethodID,Type, Creator, CreateTime)
			SELECT	A.[Year], A.GoalPeriodID, 72, 
					A.LSCompanyID, A.LSLevel1ID, A.LSLevel2ID, A.LSLevel3ID, A.LSLevel4ID, 
					A.LSLevel5ID, A.LSLevel6ID, A.LSLevel7ID, A.LSLevel8ID,  A.LSLevel9ID,
					A.LSTrackingMethodID, 2, @UserGroupID, @dGetDateTime
			FROM	#tmpTaoPhongBan (NOLOCK) A

			INSERT INTO PM_tblThietLapMucTieuCoCau_GoalCategory(ThietLapMucTieuCoCauID, GoalCategoryID, TyTrong, TyTrongCongTy)
			SELECT  D.ThietLapMucTieuCoCauID, B.GoalCategoryID, B.TyTrong, B.TyTrong
			FROM	#tmpTaoPhongBan (NOLOCK) A
					INNER JOIN PM_tblThietLapMucTieuCoCau (NOLOCK) D ON D.GoalPeriodID = A.GoalPeriodID 
						AND D.LSCompanyID = A.LSCompanyID 
						AND D.LSLevel1ID = A.LSLevel1ID
						AND D.LSLevel2ID = A.LSLevel2ID
						AND D.LSLevel3ID = A.LSLevel3ID
						AND	CASE WHEN D.LSLevel4ID IS NULL THEN -1 ELSE D.LSLevel4ID END = CASE WHEN A.LSLevel4ID IS NULL THEN -1 ELSE A.LSLevel4ID END
						AND	CASE WHEN D.LSLevel5ID IS NULL THEN -1 ELSE D.LSLevel5ID END = CASE WHEN A.LSLevel5ID IS NULL THEN -1 ELSE A.LSLevel5ID END
						AND	CASE WHEN D.LSLevel6ID IS NULL THEN -1 ELSE D.LSLevel6ID END = CASE WHEN A.LSLevel6ID IS NULL THEN -1 ELSE A.LSLevel6ID END
						AND	CASE WHEN D.LSLevel7ID IS NULL THEN -1 ELSE D.LSLevel7ID END = CASE WHEN A.LSLevel7ID IS NULL THEN -1 ELSE A.LSLevel7ID END
						AND	CASE WHEN D.LSLevel8ID IS NULL THEN -1 ELSE D.LSLevel8ID END = CASE WHEN A.LSLevel8ID IS NULL THEN -1 ELSE A.LSLevel8ID END
						AND	CASE WHEN D.LSLevel9ID IS NULL THEN -1 ELSE D.LSLevel9ID END = CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END
						AND	CASE WHEN D.LSTrackingMethodID IS NULL THEN -1 ELSE D.LSTrackingMethodID END = CASE WHEN A.LSTrackingMethodID IS NULL THEN -1 ELSE A.LSTrackingMethodID END
						AND D.Type = 2 
						AND D.Year = A.Year
					INNER JOIN PM_tblThietLapMucTieuCoCau_GoalCategory (NOLOCK) B ON B.ThietLapMucTieuCoCauID = A.ThietLapMucTieuCoCauID
					left join dbo.PM_tblGoalCategory (Nolock) GC on GC.PM_GoalCategoryID = B.GoalCategoryID
			WHERE	B.TyTrong > 0
			ORDER BY D.ThietLapMucTieuCoCauID, GC.Rank
		END
	END

	

	SELECT  'ThietLapMucTieuCoCauID'= ISNULL(A.ThietLapMucTieuCoCauID,-1),
			'SetupManagerAndCEOID'  = B.ID,
			'GoalPeriodID'			= C.PM_GoalPeriodID,
			'GoalPeriodCode'		= C.PM_GoalPeriodCode, 
			'GoalPeriodName'		= CASE WHEN @LanguageID = 'VN' THEN C.PM_GoalPeriodVNName ELSE C.PM_GoalPeriodName END,
			'GoalPeriodYear'		= C.PM_GoalPeriodYear,
			'StartDatePhongBan'		= C.StartDate_PhongBan,
			'EndDatePhongBan'		= C.EndDate_PhongBan,
			'ApprovedStatus'        = CASE WHEN @LanguageID = 'VN' THEN TT.VNName ELSE TT.Name END,
			'StatusColor'			= SC.StatusColor,
			'CompanyID'				= A.LSCompanyID,
			'Level1ID'				= A.LSLevel1ID,
			'Level2ID'				= A.LSLevel2ID,
			'Level3ID'				= A.LSLevel3ID,
			'Level4ID'				= A.LSLevel4ID,
			'Level5ID'				= A.LSLevel5ID,
			'Level6ID'				= A.LSLevel6ID,
			'Level7ID'				= A.LSLevel7ID,
			'Level8ID'				= A.LSLevel8ID,
			'Level9ID'				= A.LSLevel9ID,
			'CompanyName'			= CASE WHEN @LanguageID = 'VN' THEN Com.VNName ELSE Com.Name END, 
			'Level1Name'			= CASE WHEN @LanguageID = 'VN' THEN LV1.VNName ELSE LV1.Name END,
			'Level2Name'			= CASE WHEN @LanguageID = 'VN' THEN LV2.VNName ELSE LV2.Name END,
			'Level3Name'			= CASE WHEN @LanguageID = 'VN' THEN LV3.VNName ELSE LV3.Name END,
			'Level4Name'			= CASE WHEN @LanguageID = 'VN' THEN LV4.VNName ELSE LV4.Name END,
			'Level5Name'			= CASE WHEN @LanguageID = 'VN' THEN LV5.VNName ELSE LV5.Name END,
			'Level6Name'			= CASE WHEN @LanguageID = 'VN' THEN LV6.VNName ELSE LV6.Name END,
			'Level7Name'			= CASE WHEN @LanguageID = 'VN' THEN LV7.VNName ELSE LV7.Name END,
			'Level8Name'			= CASE WHEN @LanguageID = 'VN' THEN LV8.VNName ELSE LV8.Name END,
			'Level9Name'			= CASE WHEN @LanguageID = 'VN' THEN LV9.VNName ELSE LV9.Name END,
			'LSTrackingMethodID'	= A.LSTrackingMethodID,
			'TrackingMethod'		= CASE WHEN @LanguageID = 'VN' THEN O.VNName ELSE O.Name END,
			'TyTrongLoaiMucTieu'    = REPLACE(STUFF(
									   (
										   SELECT  ',' +  ISNULL(case when @LanguageID = 'VN' then GC.VNName else GC.Name end, '')  
													   + ' - ' + GC.PM_GoalCategoryCode + ': '
													   + dbo.fnConvertNumber(t1.TyTrong) + '%'
										   FROM		[dbo].[PM_tblThietLapMucTieuCoCau_GoalCategory] t1
													left join dbo.PM_tblGoalCategory (Nolock) GC on GC.PM_GoalCategoryID = t1.GoalCategoryID
										   WHERE	t1.ThietLapMucTieuCoCauID = A.ThietLapMucTieuCoCauID
													and t1.TyTrong > 0
											ORDER BY GC.Rank
										  FOR XML PATH('')
										), 1, 1, ''),',',CHAR(13) + CHAR(10)),
			'SoLuongLoaiMucTieu'    = REPLACE(STUFF(
									   (
										   SELECT  ',' +  ISNULL(case when @LanguageID = 'VN' then GC.VNName else GC.Name end, '')  
													   + ' - ' + GC.PM_GoalCategoryCode + ': '
													   + dbo.fnConvertNumber(t1.SoLuongMucTieu) 
										   FROM		[dbo].[PM_tblThietLapMucTieuCoCau_GoalCategory] t1
													left join dbo.PM_tblGoalCategory (Nolock) GC on GC.PM_GoalCategoryID = t1.GoalCategoryID
										   WHERE	t1.ThietLapMucTieuCoCauID = A.ThietLapMucTieuCoCauID
													and t1.SoLuongMucTieu > 0
											ORDER BY GC.Rank
										  FOR XML PATH('')
										), 1, 1, ''),',',CHAR(13) + CHAR(10)),
			'TruongBoPhan'			= E.EmpName,
			'NguoiNhapHo'			= E1.EmpName,
			'Creater'				= A.Creator,
			'CreateTime'			= CONVERT(NVARCHAR(10),A.CreateTime,103) + ' ' + CONVERT(NVARCHAR(10),A.CreateTime,108),
			'Editer'				= A.Editor,
			'EditTime'				= CONVERT(NVARCHAR(10),A.EditTime,103) + ' ' + CONVERT(NVARCHAR(10),A.EditTime,108),
			'IsEnable'				= dbo.PM_fnCheckLockThietLapMucTieuCoCau(A.ThietLapMucTieuCoCauID, 2),
			'ApprovedStatusID'		= A.ApprovedStatusID,
			--'IsEnablePhanBo'		= CASE WHEN A.ApprovedStatusID IN (78, 80) 
			--									then 1 ELSE 0 END,
			'IsEnablePhanBo'		= 1,
			'NguoiPheDuyet_Cap1'	= ISNULL(dbo.Common_fnLayNguoiPheDuyet_PassOrg(A.LSCompanyID,A.LSLevel1ID,A.LSLevel2ID,A.LSLevel3ID,A.LSLevel4ID,A.LSLevel5ID,A.LSLevel6ID,A.LSLevel7ID,A.LSLevel8ID,A.LSLevel9ID,NULL,NULL,NULL,NULL,NULL,NULL,NULL,NULL,NULL,NULL,1,1,112),''),
			'TinhTrangPhanBo'       = CASE WHEN @LanguageID = 'VN' THEN TTPB.VNName ELSE TTPB.Name END,
			'StartDatePhanBoMucTieuPhongBan'		= C.StartDate_PhanBoMucTieuPhongBan,
			'EndDatePhanBoMucTieuPhongBan'			= C.EndDate_PhanBoMucTieuPhongBan,
			'DuLieuPhanBo'			= CASE WHEN PB.ThietLapMucTieuCoCau_PhanBoMucTieuID IS NULL THEN 0 ELSE 1 END
	FROM	PM_tblThietLapMucTieuCoCau (NOLOCK) A
			OUTER APPLY  
			(
				SELECT	Top 1 B.ID, B.EmpID, B.NguoiNhapHo_EmpID
				FROM	PM_tblSetupManagerAndCEO (NOLOCK) B
				WHERE	B.Type = 2
						AND B.EffectiveDate <= GetDate()
						AND YEAR(B.EffectiveDate) = A.Year
						AND B.LSCompanyID = A.LSCompanyID
						AND B.LSLevel1ID = A.LSLevel1ID
						AND B.LSLevel2ID = A.LSLevel2ID
						AND B.LSLevel3ID = A.LSLevel3ID
						AND	CASE WHEN B.LSLevel4ID IS NULL THEN -1 ELSE B.LSLevel4ID END = CASE WHEN A.LSLevel4ID IS NULL THEN -1 ELSE A.LSLevel4ID END
						AND	CASE WHEN B.LSLevel5ID IS NULL THEN -1 ELSE B.LSLevel5ID END = CASE WHEN A.LSLevel5ID IS NULL THEN -1 ELSE A.LSLevel5ID END
						AND	CASE WHEN B.LSLevel6ID IS NULL THEN -1 ELSE B.LSLevel6ID END = CASE WHEN A.LSLevel6ID IS NULL THEN -1 ELSE A.LSLevel6ID END
						AND	CASE WHEN B.LSLevel7ID IS NULL THEN -1 ELSE B.LSLevel7ID END = CASE WHEN A.LSLevel7ID IS NULL THEN -1 ELSE A.LSLevel7ID END
						AND	CASE WHEN B.LSLevel8ID IS NULL THEN -1 ELSE B.LSLevel8ID END = CASE WHEN A.LSLevel8ID IS NULL THEN -1 ELSE A.LSLevel8ID END
						AND	CASE WHEN B.LSLevel9ID IS NULL THEN -1 ELSE B.LSLevel9ID END = CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END
				ORDER BY B.EffectiveDate desc
			) AS B 
			LEFT JOIN PM_tblGoalPeriod (NOLOCK) C ON C.PM_GoalPeriodID  = A.GoalPeriodID
			LEFT JOIN dbo.LS_tblCompany (NOLOCK) Com ON Com.LSCompanyID = A.LSCompanyID
			LEFT JOIN dbo.LS_tblLevel1 (NOLOCK) LV1 ON LV1.LSLevel1ID = A.LSLevel1ID
			LEFT JOIN dbo.LS_tblLevel2 (NOLOCK) LV2 ON LV2.LSLevel2ID = A.LSLevel2ID
			LEFT JOIN dbo.LS_tblLevel3 (NOLOCK) LV3 ON LV3.LSLevel3ID = A.LSLevel3ID
			LEFT JOIN dbo.LS_tblLevel4 (NOLOCK) LV4 ON LV4.LSLevel4ID = A.LSLevel4ID
			LEFT JOIN dbo.LS_tblLevel5 (NOLOCK) LV5 ON LV5.LSLevel5ID = A.LSLevel5ID
			LEFT JOIN dbo.LS_tblLevel6 (NOLOCK) LV6 ON LV6.LSLevel6ID = A.LSLevel6ID
			LEFT JOIN dbo.LS_tblLevel7 (NOLOCK) LV7 ON LV7.LSLevel7ID = A.LSLevel7ID
			LEFT JOIN dbo.LS_tblLevel8 (NOLOCK) LV8 ON LV8.LSLevel8ID = A.LSLevel8ID
			LEFT JOIN dbo.LS_tblLevel9 (NOLOCK) LV9 ON LV9.LSLevel9ID = A.LSLevel9ID
			LEFT JOIN dbo.PM_tblApprovedStatus (Nolock) TT on TT.PM_ApprovedStatusID = CASE WHEN A.ApprovedStatusID is null THEN 72 ELSE A.ApprovedStatusID END
			LEFT JOIN dbo.LS_tblStatusColor (Nolock) SC on SC.StatusColorID = TT.StatusColorID
			LEFT JOIN dbo.HR_vEmp (NOLOCK) E ON E.EmpID = B.EmpID
			LEFT JOIN dbo.HR_vEmp (NOLOCK) E1 ON E1.EmpID = B.NguoiNhapHo_EmpID
			LEFT JOIN dbo.LS_tblTrackingMethod (NOLOCK) O ON O.LSTrackingMethodID = A.LSTrackingMethodID
			--left join dbo.fnGetAllLevel_Permission(@UserGroupID,@FunctionID,'Comp') Co on A.LSCompanyID = Co.strValue

							AND	CASE WHEN B.LSLevel9ID IS NULL THEN -1 ELSE B.LSLevel9ID END = CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END
							AND	CASE WHEN B.LSTrackingMethodID IS NULL THEN -1 ELSE B.LSTrackingMethodID END = CASE WHEN A.LSTrackingMethodID IS NULL THEN -1 ELSE A.LSTrackingMethodID END
					ORDER BY EffectiveDate desc
				) AS B 
				left join dbo.HR_vEmp (Nolock) CEO on CEO.EmpID = B.EmpID	
				left join dbo.HR_vEmp (Nolock) CEONhapHo on CEONhapHo.EmpID = B.NguoiNhapHo_EmpID
