				exec [dbo].[TS_spfrmLogData_Save] @UserID=@UserGroupID,@TableName='TER_TBLFINALPAYMENTLog',@Action='Update',@IsLogNew=1,@Xml=@Xml,@IDLog=@IDLog
			end
		END	
	END

ELSE IF @Activity = 'LoadNgayHopLe' 
BEGIN		
	Select top(1) 
		@QTTV_TuNgay_NgayVaoLam = QTTV_TuNgay_NgayVaoLam, 
		@QTTV_TuNgay_NgayVaoChinhThuc = QTTV_TuNgay_NgayVaoChinhThuc,
		@QTTV_TuNgay_Date1 = QTTV_TuNgay_Date1
	From TER_tblSetupTER (NoLock)
		
	If isnull(@QTTV_TuNgay_NgayVaoLam,0) = 0 and isnull(@QTTV_TuNgay_NgayVaoChinhThuc,0)=0 and isnull(@QTTV_TuNgay_Date1, 0) = 0
	Begin
		Set @QTTV_TuNgay_NgayVaoLam = 1
	End
				
		if not exists(Select top 1 1 From [dbo].[TER_tblFinalPaymentRecord](nolock) Where EmpID = @EmpID)
		Begin
			SELECT @dFromDate = isnull(NgayTinhTroCap, Case when @QTTV_TuNgay_NgayVaoLam = 1 then StartDate
															when @QTTV_TuNgay_NgayVaoChinhThuc = 1 then StartDateCompany 
															--BaoNTQ ngày 18/07/2013
															when @QTTV_TuNgay_Date1 = 1 then 															
																							case when isnull(C.Date1, '') = '' then StartDate else C.Date1 end
																						end 
															--End
										)
			FROM dbo.HR_tblEmp (Nolock) A
				left join HR_tblEmpOther(nolock) B on A.EmpID = B.EmpID
				left join HR_TBLEMPCV(nolock) C on A.EmpID = C.EmpID
			WHERE A.EmpID = @EmpID
		End
		Else
		Begin
			--Declare @NgayQLLam_New datetime, @NgayTV_New datetime, @NgayTV_GanNhat datetime
			set @NgayQLLam_New = null
			set @NgayTV_New = null
			set @NgayTV_GanNhat = null

			--Lay ra ngay thoi viec
			Select @NgayTV_New = EffectDate
			From TER_tblTermination(nolock)
			Where EmpID = @EmpID
			--Khong quan tam hinh` thuc la` gi`
			--Lay ngay thoi viec gan nhat
			Select @NgayTV_GanNhat = max(FromDate)
			From HR_tblWorkingRecord(nolock) A
			left join [dbo].[LS_tblStatusChange](nolock) B on A.LSStatusChangeID= B.LSStatusChangeID
			Where EmpID = @EmpID And FromDate < @NgayTV_New And B.LSStatusChangeCode = 'RES'
				
			If @NgayTV_GanNhat is not null
			Begin
				--Lay ngay lam viec lien ke
				Select Top 1 @NgayQLLam_New = FromDate
				From HR_tblWorkingRecord(nolock) A
				Where FromDate >= @NgayTV_GanNhat
				Order by FromDate Asc

				Select @dFromDate = @NgayQLLam_New
			End
			Else
			Begin
				SELECT @dFromDate = isnull(NgayTinhTroCap, Case when @QTTV_TuNgay_NgayVaoLam = 1 then StartDate
															when @QTTV_TuNgay_NgayVaoChinhThuc = 1 then StartDateCompany 
															--BaoNTQ ngày 18/07/2013
															when @QTTV_TuNgay_Date1 = 1 then 															
																							case when isnull(C.Date1, '') = '' then StartDate else C.Date1 end
																						end 
															--End
										)
				FROM dbo.HR_tblEmp (Nolock) A
					left join HR_tblEmpOther(nolock) B on A.EmpID = B.EmpID
					left join HR_TBLEMPCV(nolock) C on A.EmpID = C.EmpID
				WHERE A.EmpID = @EmpID
			End
		End
		
		SELECT @dLastWorkingDate = LastWorkingDate FROM Ter_tblTermination WHERE EmpID = @EmpID
		IF (@dFromDate > CONVERT(DATETIME, '31/12/2008', 103))
			SET @dFromDate = CONVERT(DATETIME, '31/12/2008', 103)
		IF (@dLastWorkingDate > CONVERT(DATETIME, '31/12/2008', 103))
			SET @dToDate = CONVERT(DATETIME, '31/12/2008', 103)
		ELSE 
			SET @dToDate = @dLastWorkingDate
			
		SELECT CONVERT(NVARCHAR(10), @dFromDate, 103) AS FromDate,
				CONVERT(NVARCHAR(10), @dToDate, 103) AS ToDate
END
