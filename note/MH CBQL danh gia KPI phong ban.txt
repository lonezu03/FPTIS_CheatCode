	if object_id('tempdb..#DanhGiaMucTieuCoCau_Transfer') is not null 
		drop table dbo.[#DanhGiaMucTieuCoCau_Transfer]

	Create table dbo.[#DanhGiaMucTieuCoCau_Transfer]
	(
		DanhGiaMucTieuCoCauID	int null
	) 	
	
	INSERT INTO dbo.[#DanhGiaMucTieuCoCau_Transfer]
	(
		DanhGiaMucTieuCoCauID
	)
	SELECT	A.DanhGiaMucTieuCoCauID	
	FROM   
	(
		SELECT case when MasterID = '' then null else MasterID end as DanhGiaMucTieuCoCauID
		FROM OPENXML(@handle, '/Root/Record', 2) WITH 
		( 
			MasterID nvarchar(12)
		)
	) A

	EXEC sp_xml_removedocument @handle
	
	SET @dGetDateTime = GETDATE()
	IF EXISTS (SELECT TOP 1 1 FROM #DanhGiaMucTieuCoCau_Transfer(NOLOCK))
	BEGIN
		DECLARE ChuyenDanhGia CURSOR
		FOR 
			SELECT	A.DanhGiaMucTieuCoCauID
			FROM	#DanhGiaMucTieuCoCau_Transfer(NOLOCK) A
		OPEN ChuyenDanhGia
		FETCH NEXT FROM ChuyenDanhGia INTO @Cur_DanhGiaMucTieuCoCauID
		WHILE @@FETCH_STATUS = 0 
		BEGIN
			SELECT	@LSCompanyID = LSCompanyID,
					@LSLevel1ID	 = LSLevel1ID,
					@LSLevel2ID	 = LSLevel2ID,
					@LSLevel3ID	 = LSLevel3ID,
					@LSLevel4ID	 = LSLevel4ID,
					@LSLevel5ID	 = LSLevel5ID,
					@LSLevel6ID	 = LSLevel6ID,
					@LSLevel7ID	 = LSLevel7ID,
					@LSLevel8ID	 = LSLevel8ID,
					@LSLevel9ID	 = LSLevel9ID
			FROM	dbo.PM_tblDanhGiaMucTieuCoCau(NoLock) 
			WHERE	DanhGiaMucTieuCoCauID = @Cur_DanhGiaMucTieuCoCauID

			SET @Approver = ISNULL(dbo.Common_fnLayNguoiPheDuyet_PassOrg(@LSCompanyID,@LSLevel1ID,@LSLevel2ID,@LSLevel3ID,@LSLevel4ID,@LSLevel5ID,@LSLevel6ID,@LSLevel7ID,@LSLevel8ID,@LSLevel9ID,NULL,NULL,NULL,NULL,NULL,NULL,NULL,NULL,NULL,NULL,1,1,113),'')

			IF ISNULL(@Approver,'') <> ''
			BEGIN
				SET @ReturnMess = dbo.SYS_fnGetMess('Approve_002', @LanguageID)
			END
				
			UPDATE	dbo.PM_tblDanhGiaMucTieuCoCau
			SET		ApprovedLevelID = 1,
					CurApprover = @Approver,
					ApprovedStatusID = 86,
					LyDoTuChoi = NULL,
					Editor = @UserGroupID,
					EditTime = @dGetDateTime
			WHERE	DanhGiaMucTieuCoCauID = @Cur_DanhGiaMucTieuCoCauID

			INSERT INTO dbo.PM_tblDanhGiaMucTieuCoCauProcess(DanhGiaMucTieuCoCauID, ApprovedLevelID, CurApprover, ApprovedStatusID, Creater, CreateTime)
			VALUES  (@Cur_DanhGiaMucTieuCoCauID, 1, @Approver, 86, @UserGroupID, @dGetDateTime)
            
			FETCH NEXT FROM ChuyenDanhGia INTO @Cur_DanhGiaMucTieuCoCauID
		END
		CLOSE ChuyenDanhGia
		DEALLOCATE ChuyenDanhGia
	END
	SELECT	@Return_ID		= ISNULL(@Return_ID,'')			+ CONVERT(nvarchar(12),A.DanhGiaMucTieuCoCauID)	+ '$',
			@Return_EmpID	= ISNULL(@Return_EmpID,'')		+ CONVERT(nvarchar(12),B.EmpID)						+ '$',
			@Return_Cap		= ISNULL(@Return_Cap,'')		+ CONVERT(nvarchar(12),0)							+ '$',
			@Return_Action	= ISNULL(@Return_Action,'')		+ 'CHUYEN'											+ '$'
	FROM	dbo.[#DanhGiaMucTieuCoCau_Transfer] (Nolock) A
			LEFT JOIN dbo.PM_tblDanhGiaMucTieuCoCau (Nolock) B on B.DanhGiaMucTieuCoCauID = A.DanhGiaMucTieuCoCauID
			
	SET @ReturnID = @Return_ID + '#' + @Return_EmpID + '#' + @Return_Cap + '#' + @Return_Action	
	SET @ReturnMess = '0046'
	SET @ReturnMessCode = '1'
END
ELSE IF @Activity = 'GetListMail'
BEGIN
	SELECT		'; ' + ISNULL(CEO.Email, '') + '; ' + ISNULL(CEONhapHo.Email, '') as Mail
	FROM		PM_tblDanhGiaMucTieuCoCau (NOLOCK) A
				OUTER APPLY  
				(
					SELECT	Top 1 B.ID, B.EmpID, B.NguoiNhapHo_EmpID
					FROM	PM_tblSetupManagerAndCEO (NOLOCK) B
							LEFT JOIN PM_tblEvaluationPeriod (NOLOCK) C ON C.PM_EvaluationPeriodID  = A.EvaluationPeriodID
					WHERE	B.Type = A.Type
							AND B.EffectiveDate <= GetDate()
							AND YEAR(B.EffectiveDate) = A.Year
							AND B.LSCompanyID = A.LSCompanyID
							AND B.LSLevel1ID = A.LSLevel1ID
							AND B.LSLevel2ID = A.LSLevel2ID
							AND CASE WHEN B.LSLevel3ID IS NULL THEN -1 ELSE B.LSLevel3ID END = CASE WHEN A.LSLevel3ID IS NULL THEN -1 ELSE A.LSLevel3ID END
							AND CASE WHEN B.LSLevel4ID IS NULL THEN -1 ELSE B.LSLevel4ID END = CASE WHEN A.LSLevel4ID IS NULL THEN -1 ELSE A.LSLevel4ID END
							AND CASE WHEN B.LSLevel5ID IS NULL THEN -1 ELSE B.LSLevel5ID END = CASE WHEN A.LSLevel5ID IS NULL THEN -1 ELSE A.LSLevel5ID END
							AND CASE WHEN B.LSLevel6ID IS NULL THEN -1 ELSE B.LSLevel6ID END = CASE WHEN A.LSLevel6ID IS NULL THEN -1 ELSE A.LSLevel6ID END
							AND CASE WHEN B.LSLevel7ID IS NULL THEN -1 ELSE B.LSLevel7ID END = CASE WHEN A.LSLevel7ID IS NULL THEN -1 ELSE A.LSLevel7ID END
							AND CASE WHEN B.LSLevel8ID IS NULL THEN -1 ELSE B.LSLevel8ID END = CASE WHEN A.LSLevel8ID IS NULL THEN -1 ELSE A.LSLevel8ID END
							AND CASE WHEN B.LSLevel9ID IS NULL THEN -1 ELSE B.LSLevel9ID END = CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END
							AND (
									C.PM_EvaluationPeriodTypeID = 1
									OR
									C.PM_EvaluationPeriodTypeID = 2
									OR
									(C.PM_EvaluationPeriodTypeID = 3 AND B.LSTrackingMethodID = 2)
									OR
									(C.PM_EvaluationPeriodTypeID = 5 AND B.LSTrackingMethodID = 3)
								)
					ORDER BY EffectiveDate desc
				) AS B 
				left join dbo.HR_vEmp (Nolock) CEO on CEO.EmpID = B.EmpID	
				left join dbo.HR_vEmp (Nolock) CEONhapHo on CEONhapHo.EmpID = B.NguoiNhapHo_EmpID	
	WHERE		A.DanhGiaMucTieuCoCauID = @MasterID
END
ELSE IF @Activity = 'GetListMailTBP'
BEGIN
	SELECT	TBP.Email as ToMail,
			ISNULL(TBPNhapHo.Email, '') as CCMail
	FROM	PM_tblDanhGiaMucTieuCoCau (NOLOCK) A 
			OUTER APPLY  
			(
				SELECT	Top 1 B.ID, EmpID, NguoiNhapHo_EmpID
				FROM	PM_tblSetupManagerAndCEO (NOLOCK) B
						LEFT JOIN PM_tblEvaluationPeriod (NOLOCK) C ON C.PM_EvaluationPeriodID  = A.EvaluationPeriodID
				WHERE	B.Type = A.Type
						AND B.EffectiveDate <= GetDate()
						AND YEAR(B.EffectiveDate) = A.Year
						AND A.LSCompanyID = B.LSCompanyID
						AND A.LSLevel1ID = B.LSLevel1ID
						AND A.LSLevel2ID = B.LSLevel2ID
						AND CASE WHEN A.LSLevel3ID IS NULL THEN -1 ELSE A.LSLevel3ID END = CASE WHEN B.LSLevel3ID IS NULL THEN -1 ELSE B.LSLevel3ID END
						AND	CASE WHEN A.LSLevel4ID IS NULL THEN -1 ELSE A.LSLevel4ID END = CASE WHEN B.LSLevel4ID IS NULL THEN -1 ELSE B.LSLevel4ID END
						AND	CASE WHEN A.LSLevel5ID IS NULL THEN -1 ELSE A.LSLevel5ID END = CASE WHEN B.LSLevel5ID IS NULL THEN -1 ELSE B.LSLevel5ID END
						AND	CASE WHEN A.LSLevel6ID IS NULL THEN -1 ELSE A.LSLevel6ID END = CASE WHEN B.LSLevel6ID IS NULL THEN -1 ELSE B.LSLevel6ID END
						AND	CASE WHEN A.LSLevel7ID IS NULL THEN -1 ELSE A.LSLevel7ID END = CASE WHEN B.LSLevel7ID IS NULL THEN -1 ELSE B.LSLevel7ID END
						AND	CASE WHEN A.LSLevel8ID IS NULL THEN -1 ELSE A.LSLevel8ID END = CASE WHEN B.LSLevel8ID IS NULL THEN -1 ELSE B.LSLevel8ID END
						AND	CASE WHEN A.LSLevel9ID IS NULL THEN -1 ELSE A.LSLevel9ID END = CASE WHEN B.LSLevel9ID IS NULL THEN -1 ELSE B.LSLevel9ID END
						AND (
								C.PM_EvaluationPeriodTypeID = 1
								OR
								C.PM_EvaluationPeriodTypeID = 2
								OR
								(C.PM_EvaluationPeriodTypeID = 3 AND B.LSTrackingMethodID = 2)
								OR
								(C.PM_EvaluationPeriodTypeID = 5 AND B.LSTrackingMethodID = 3)
							)
				ORDER BY EffectiveDate desc
			) AS B 
			left join dbo.HR_vEmp (Nolock) TBP on TBP.EmpID = B.EmpID	
			left join dbo.HR_vEmp (Nolock) TBPNhapHo on TBPNhapHo.EmpID = B.NguoiNhapHo_EmpID	
	WHERE	A.DanhGiaMucTieuCoCauID = @MasterID
END
